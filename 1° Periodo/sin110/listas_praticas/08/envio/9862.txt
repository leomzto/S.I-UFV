#include <stdio.h>

long hiper_factorial(int n);

int main(void)
{
    int n;
    printf("Digite um numero inteiro: ");
    scanf("%d", &n);

    long result = hiper_factorial(n);
    printf("Hiper-fatorial de %d: %ld\n", n, result);

    return 0;
}

long hiper_factorial(int n)
{
    if (n <= 1)
        return 1;
    else
    {
        long pow = 1;
        for (int i = 0; i < n; i++)
            pow *= n;
        return pow * hiper_factorial(n - 1);
    }
}

#include <stdio.h>

int mdc(int a, int b);

int main(void)
{
    int a, b;
    printf("Digite dois numeros inteiros: ");
    scanf("%d %d", &a, &b);

    int result = mdc(a, b);
    printf("MDC de %d e %d: %d\n", a, b, result);

    return 0;
}

int mdc(int a, int b)
{
    if (b == 0)
        return a;
    else
        return mdc(b, a % b);
}

#include <stdio.h>

int retornaMaior(int vetor[], int tamanho);

int main(void)
{
    int vetor[] = {1, 2, 3};
    int tamanho = sizeof(vetor) / sizeof(vetor[0]);
    
    int maior = retornaMaior(vetor, tamanho);
    printf("O maior numero do vetor e: %d\n", maior);
    
    return 0;
}

int retornaMaior(int vetor[], int tamanho)
{
    if (tamanho == 1)
        return vetor[0];
    else
    {
        int maior = retornaMaior(vetor, tamanho - 1);
        return (vetor[tamanho - 1] > maior) ? vetor[tamanho - 1] : maior;
    }
}

#include <stdio.h>

void trocar(int *a, int *b);

int main(void)
{
    int a, b;
    printf("Digite dois numeros inteiros: ");
    scanf("%d %d", &a, &b);

    printf("Antes da troca: a = %d, b = %d\n", a, b);
    trocar(&a, &b);
    printf("Depois da troca: a = %d, b = %d\n", a, b);

    return 0;
}

void trocar(int *a, int *b)
{
    int temp = *a; 
    *a = *b;      
    *b = temp;
}

{
